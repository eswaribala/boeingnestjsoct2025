npm i -g @nestjs/cli
nest --version
# Install Nx globally
npm install -g nx
From node js cmd
 npm i -g pnpm
 corepack enable
corepack prepare pnpm@9 --activate
pnpm -v
1) Create Nx Workspace (Nest preset)
npx --yes create-nx-workspace@latest boeing-banking --preset=nest --appName=api-gateway --workspaceType=integrated --packageManager=npm --e2eTestRunner=none --unitTestRunner=jest --nxCloud=skip --skipGit --interactive=false
2) run api gateway
#be @ boeing-baking
npx nx serve api-gateway
3) to show the projects
npx nx show projects
4) create customer-api
npx nx g @nx/nest:app customer-api --e2eTestRunner=none
5) scaffolding
Scaffold the “customers” feature
cd customer-api
npx -p @nestjs/cli nest g module app/customers
npx -p @nestjs/cli nest g service app/customers --flat
npx -p @nestjs/cli nest g controller app/customers --flat
